\chapter{TeorĂ­a de la computabilidad}
\section{Computabilidad bĂˇsica}
\subsection{Primitivo-recursividad}
\nomenclature{$\omath$}{El conjunto de los nĂşmeros naturales.}

\begin{definition}
    Se define al conjunto de funciones sobre $\omega$ como sigue:
    \[
        \mfunc\coloneqq\set{f:\omath[n]\to\omath|n\in\omath[+]}
    \]
    Sea $n\in\omath$, entonces denotaremos por $\mfunc[n]$ al conjunto de funciones sobre $\omath$ de aridad $n$:
    \[
        \mfunc[n]\coloneqq\set{f\in\mfunc:\maridad{f} = n}
    \]
\end{definition}

\begin{definition}[Funciones bĂˇsicas]
    Se definen las \emph{funciones bĂˇsicas} como las siguientes funciones de $\mfunc$:
    \begin{itemize}
        \item La funciĂłn \emph{cero} $\mcero\in\mfunc[1]$ tal que para cada $n\in\omath$, $\mcero\pa{n} = 0$.
        \item La funciĂłn \emph{sucesor} $\msuc\in\mfunc[1]$ tal que para cada $n\in\omath$, $\msuc\pa{n} = n + 1$.
        \item Para cada $n, k\in\omath[+]$\nomenclature{$\omath[+]$}{El conjunto de nĂşmeros naturales positivos.} con $k\leq n$, la funciĂłn \emph{proyecciĂłn} $\mproj{n}{k}\in\mfunc[n]$ tal que para cada $\pa{x_{1},\dotsc,x_{n}}\in\omath[n]$, $\mproj{n}{k}\pa{x_{1},\dotsc,x_{n}} = x_{k}$.
    \end{itemize}
\end{definition}

\begin{definition}[ComposiciĂłn generalizada]
    Sean $n, k\in\omath$, $f\in\mfunc[n]$ y $g_{1},\dotsc, g_{n}\in\mfunc[k]$\nomenclature{$\maridad{f}$}{Aridad de la funciĂłn $f$, es decir, el nĂşmero de argumentos admitidos por la funciĂłn $f$.}. La \emph{composiciĂłn generalizada} de las funciones $f, g_{1},\dotsc, g_{n}$ es la Ăşnica funciĂłn $h\in\mfunc[k]$ tal que para cada $x\in\omath[k]$:
        \[
            h\pa{x} = f\pa{g_{1}\pa{x},\dotsc,g_{n}\pa{x}}
        \]
\end{definition}

\begin{definition}[RecursiĂłn generalizada]
    Sean $n\in\omath$, $f\in\mfunc[n]$ y $g\in\mfunc[n + 2]$. La \emph{recursiĂłn generalizada} de las funciones $f$ y $g$ es la Ăşnica funciĂłn $h:\mfunc[n + 1]$ tal que para cada $x\in\omath[n]$ y $y\in\omath$:
        \[
            \begin{array}{c}
                h\pa{x, 0} = f\pa{x}\\
                h\pa{x, \msuc\pa{y}} = g\pa{x, y, h\pa{x, y}}
            \end{array}
        \]
\end{definition}

\begin{definition}[Funciones primitivas recursivas]
    Se define el conjunto de \emph{funciones primitivas recursivas} como el mĂ­nimo conjunto $\mprim\subseteq\mfunc$ que contiene a las funciones bĂˇsicas y que es cerrado bajo composiciĂłn y recursiĂłn generalizada.

    Sea $n\in\omath$, entonces denotaremos por $\mprim[n]$ al conjunto de funciones primitivas recursivas de aridad $n$:
    \[
        \mprim[n]\coloneqq\set{f\in\mprim:\maridad{f} = n}
    \]
\end{definition}

\begin{theorem}
    Se satisface que las siguientes funciones son primitivas recursivas.
    \begin{enumerate}
        \item La funciĂłn suma $f\in\mfunc[2]$ tal que $f\pa{x, y} = x + y$.
        \item La funciĂłn producto $f\in\mfunc[2]$ tal que $f\pa{x, y} = x\cdot y$.
        \item La funciĂłn exponenciaciĂłn $f\in\mfunc[2]$ tal que $f\pa{x, y} = x^{y}$.
        \item Para cada $n\in\omath[+]$, la funciĂłn cero generalizada $f\in\mfunc[n]$ tal que $f\pa{x} = 0$.
        \item La funciĂłn predecesor $f:\in\mfunc[1]$ tal que:
        \[
            f\pa{x} =\begin{cases}
               x - 1, & x > 0\\
               0 & \text{en caso contrario}
            \end{cases}
        \]
        \item La funciĂłn resta $f\in\mfunc[2]$ tal que:
        \[
            f\pa{x} =\begin{cases}
                x - y, & x > y\\
                0, & \text{en caso contrario}
            \end{cases}
        \]
    \end{enumerate}
\end{theorem}

\begin{definition}[Conjuntos primitivos recursivos]
    Sean $n\in\omath[+]$ y $R\in\mpow{\omath[n]}$. $R$ se dice \emph{primitivo recursivo} si $\mchi{R}\in\mprim$.
\end{definition}

\begin{theorem}
    Sean $n\in\omath[+]$ y $R\in\mpow{\omath[n]}$. $R$ es primitivo recursivo si y solo si $\omath[n]\setminus R$ es primitivo recursivo.
\end{theorem}

\begin{theorem}
    Las relaciones $\set{0}$, $<$, $>$ y $=$ son primitivos recursivos.
\end{theorem}

\begin{coll}
    Las relaciones $\geq$, $\leq$ y $\neq$ son primitivos recursivos.
\end{coll}

\begin{definition}[FĂłrmulas primitivas recursivas]
    Sean $n\in\omath[+]$ y $\varphi\pa{x_{1},\dotsc,x_{n}}$ una fĂłrmula con variables libres en $\set{x_{1},\dotsc,x_{n}}$. $\varphi$ se denomina primitiva recursiva si el conjunto $\set{\pa{x_{1},\dotsc,x_{n}}\in\omath[n]:\varphi\pa{x_{1},\dotsc,x_{n}}}$ es primitivo recursivo.
\end{definition}

\begin{theorem}
    Sean $\varphi$ y $\psi$ fĂłrmulas primitivas recursivas. Entonces $\neg\varphi$, $\varphi\vee\psi$ y $\varphi\land\psi$ son primitivas recursivas.
\end{theorem}

\begin{coll}
    Sean $n\in\omath[+]$ y $X, Y\in\mpow{\omath[n]}$ primitivos recursivos. Entonces $X\cup Y$, $X\cap Y$ y $X\setminus Y$ son primitivos recursivos.
\end{coll}

\begin{theorem}
    Sean $n\in\omath[+]$, $y\in\omath$ y $f\in\mprim[n + 1]$. Entonces las siguientes funciones son primitivas recursivas:
    \begin{enumerate}
        \item $g\in\mfunc[n]$ tal que para cada $x\in\omath[n]$, $g\pa{x} =\sum\limits_{z < y}f\pa{x, z}$.
        \item $g\in\mfunc[n]$ tal que para cada $x\in\omath[n]$, $g\pa{x} =\prod\limits_{z < y}f\pa{x, z}$.
    \end{enumerate}
\end{theorem}

\begin{notation}
    Sean $\varphi$ una fĂłrmula, $x$ una variable y $t$ un tĂ©rmino que no contiene a $x$. Se denotarĂˇ a las fĂłrmulas $\exists x\pa{\pa{x < t}\land\varphi}$ y $\forall x\pa{\pa{x < t}\to\varphi}$ por $\pa{\exists x < t}\varphi$ y $\pa{\forall x < t}\varphi$ respectivamente. 
\end{notation}

\begin{theorem}
    Sean $n\in\omath[+]$, $f\in\mprim[n]$ y $R\in\mpow{\omath[n + 1]}$ una relaciĂłn primitiva recursiva. Entonces las siguientes relaciones son primitivas recursivas:
    \begin{enumerate}
        \item $\set{x\in\omath[n]:\pa{\exists y < f\pa{x}}R\pa{x, y}}$
        \item $\set{x\in\omath[n]:\pa{\forall y < f\pa{x}}R\pa{x, y}}$
    \end{enumerate}
\end{theorem}

\begin{theorem}
    Sean $n, k\in\omath[+]$, $g_{1},\dotsc, g_{n}, h\in\mprim[k]$ y $R_{1},\dotsc, R_{n}\in\mpow{\omath[k]}$ primitivas recursivas disjuntas a pares. Entonces la funciĂłn $f\in\mfunc[k]$ dada como sigue para cada $x\in\omath[k]$ es primitiva recursiva:
    \[
        f\pa{x} =\begin{cases}
            g_{1}\pa{x}, & R_{1}\pa{x}\\
            \vdots &\\
            g_{n}\pa{x}, & R_{n}\pa{x}\\
            h\pa{x}, &\text{en otro caso}
        \end{cases}
    \]
\end{theorem}

\begin{definition}[Operador $\mu$ acotado]
    Sean $n\in\omath[+]$, $R\in\mpow{\omath[n + 1]}$ y $z\in\omath$. El \emph{operador $\mu$ (acotado)} de $R$ con cota $z$ es la funciĂłn $f\in\mfunc[n]$ tal que para cada $x\in\omath[n]$:
    \[
        f\pa{x} =\begin{cases}
            \min\set{y\in\omath:\pa{y < z}\land R\pa{x, y}}, &\text{si existe}\\
            0, & \text{en otro caso}
        \end{cases}
    \]
    A tal funciĂłn se le denota por $\pa{\mu y < z}R\pa{x, y}$.
\end{definition}

\begin{theorem}
    Sean $n\in\omath$, $g\in\mprim[n]$ y $R\in\mpow{\omath[n + 1]}$ primitiva recursiva. Entonces la funciĂłn $f\in\mfunc[n]$ tal que para cada $x\in\omath[n]$, $f(x) =\pa{\mu y < g(x)}R(x, y)$ es primitiva recursiva.
\end{theorem}

\subsection{Parcial-recursividad}

\begin{definition}[Funciones parciales]
    Sea $\partial$ un elemento que no pertenece a $\omath$. Definimos al conjunto $\omath[\partial]\coloneqq\omath\cup\set{\partial}$.
    
    Se define el conjunto de \emph{funciones parciales} como sigue:
    \[
        \mfunc[][\partial]\coloneqq\set{f:\omath[n]\to\omath[\partial]| n\in\omath[+]}
    \]
    
    Para cada $n\in\omath[+]$, se define al conjunto de funciones parciales de aridad $n$ como:
    \[
        \mfunc[n][\partial]\coloneqq\set{f\in\mfunc[][\partial]:\maridad{f} = n}
    \]

    Sean $n\in\omath[+]$ y $f\in\mfunc[n][\partial]$. Para cada $x\in\omath[n]$, si $f\pa{x} =\partial$, entonces se denotarĂˇ por $f\pa{x}\partial$; y si existe $y\in\omath$ tal que $f\pa{x} = y$, entonces se denotarĂˇ por $f\pa{x}\halt$.

    Sean $n\in\omath[+]$ y $f\in\mfunc[n][\partial]$. Se define el \emph{dominio} de $f$ como el conjunto:
    \[
        \mdom{f}\coloneqq\set{x\in\omath[n]: f(x)\halt}
    \]

    Sean $n\in\omath[+]$ y $f\in\mfunc[n][\partial]$. Se define el \emph{rango} de $f$ como el conjunto:
    \[
        \mran{f}\coloneqq\set{y\in\omath:\exists x\pa{x\in\omath[n]\land\pa{f(x) = y}}}
    \]

    Sean $n\in\omath[+]$ y $f\in\mfunc[n][\partial]$. $f$ se denomina \emph{total} si $\mdom{f} =\omath[n]$.
\end{definition}

\begin{definition}[ComposiciĂłn generalizada para funciones parciales]
    Sean $n, k\in\omath[+]$, $f\in\mfunc[n][\partial]$ y $g_{1},\dotsc,g_{n}\in\mfunc[k][\partial]$. La \emph{composiciĂłn generalizada} de $f$ con $g_{1},\dotsc, g_{n}$ es la Ăşnica funciĂłn $h\in\mfunc[k][\partial]$ tal que para cada $x\in\omath[k]$:
    \[
        h(x) =\begin{cases}
            f\pa{g_{1}\pa{x},\dotsc,g_{n}\pa{x}}, &\pa{\forall k\leq n}g_{k}(x)\halt\\
            \partial, &\text{en otro caso}
        \end{cases}
    \]
\end{definition}

\begin{definition}[Recursividad generalizada para funciones parciales]
    Sean $n\in\omath[+]$, $f\in\mfunc[n][\partial]$ y $g\in\mfunc[n + 2][\partial]$. La \emph{recursiĂłn generalizada} de $f$ con $g$ es la Ăşnica funciĂłn $h\in\mfunc[n + 1][\partial]$ tal que para cada $x\in\omath[n]$ y cada $y\in\omath$.
    \[
        \begin{array}{c}
            h\pa{x, 0} = f(x)\\
            h\pa{x,\msuc\pa{y}} =\begin{cases}
                g\pa{x, y, h\pa{x, y}}, & h\pa{x, y}\halt\\
                \partial, &\text{en otro caso}
            \end{cases}
        \end{array}
    \]
\end{definition}

\begin{definition}[Operador $\mu$ no acotado]
    Sean $n\in\omath[+]$ y $R\in\mpow{\omath[n + 1]}$. El \emph{operador $\mu$ (no acotado)} de $R$ es la funciĂłn $f\in\mfunc[n][\partial]$ tal que para cada $x\in\omath[n]$:
    \[
        f\pa{x} =\begin{cases}
            \min\set{y\in\omath:R\pa{x, y}}, &\text{si existe}\\
            \partial, & \text{en otro caso}
        \end{cases}
    \]
    A tal funciĂłn se le denota por $\mu y R\pa{x, y}$.
\end{definition}

\begin{definition}[Funciones parcial recursivas]
    El conjunto de \emph{funciones parcial recursivas} es el mĂ­nimo conjunto $\mpartial\subseteq\mfunc[][\partial]$ que contiene las funciones bĂˇsicas, es cerrado bajo composiciĂłn y recursividad general de funciones parciales, y ademĂˇs satisface que para cada $n\in\omath[+]$ y cada $g\in\mpartial$ con $\maridad{g} = n + 1$, definiĂ©ndose la relaciĂłn $R =\set{\pa{x, y}\in\omath[n + 1]: \pa{g(x, y) = 0}\land\pa{\forall z < y}g(x, y)\halt}$, se cumple que la funciĂłn $f\in\mfunc[n][\partial]$ tal que $f(x) =\mu yR(x, y)$ para cada $x\in\omath[n]$ estĂˇ en $\mpartial$.

    Definimos al conjunto $\mpartial[n]\coloneqq\set{f\in\mpartial:\maridad{f} = n}$.

    Sean $n\in\omath[+]$ y $f\in\mpartial[n]$. $f$ se dice \emph{total recursiva} si es total.

    Se define el conjunto de funciones total recursivas como $\mtotal\coloneqq\set{f\in\mpartial: f\text{ es total}}$, y para cada $n\in\omath[+]$, se define $\mtotal[n]\coloneqq\set{f\in\mtotal:\maridad{f} = n}$.
\end{definition}

\begin{theorem}
    $\mprim\subsetneq\mpartial$
\end{theorem}

\begin{definition}[Conjunto recursivo]
    Sean $n\in\omath[+]$ y $R\in\mpow{\omath[n]}$. $R$ se denomina \emph{recursivo} si $\mchi{R}\in\mtotal$.
\end{definition}

\begin{theorem}
    Sean $n\in\omath[+]$ y $R\in\mpow{\omath[n]}$. $R$ es recursivo si y solo si $\omath[n]\setminus R$ es recursivo.
\end{theorem}

\begin{definition}
    Sean $n\in\omath[+]$ y $\varphi\pa{x_{1},\dotsc,x_{n}}$ una fĂłrmula con variables libres en $\set{x_{1},\dotsc,x_{n}}$. $\varphi$ se denomina recursiva si el conjunto $\set{\pa{x_{1},\dotsc,x_{n}}\in\omath[n]:\varphi\pa{x_{1},\dotsc,x_{n}}}$ es recursivo.
\end{definition}

\begin{theorem}
    Sean $\varphi$ y $\psi$ fĂłrmulas recursivas. Entonces $\neg\varphi$, $\varphi\vee\psi$ y $\varphi\land\psi$ son recursivas.
\end{theorem}

\begin{coll}
    Sean $n\in\omath[+]$ y $X, Y\in\mpow{\omath[n]}$ recursivos. Entonces $X\cup Y$, $X\cap Y$ y $X\setminus Y$ son recursivos.
\end{coll}

\begin{theorem}
    Sean $n\in\omath[+]$ y $R\in\mpow{\omath[n + 1]}$ recursiva. Entonces la funciĂłn $f\in\mfunc[n][\partial]$ tal que para cada $x\in\omath[n]$, $f(x) =\mu yR(x, y)$ es parcial recursiva.
\end{theorem}

\begin{theorem}
    Sean $n\in\omath[+]$ y $X\in\mpow{\omath}$ recursivo infinito. Entonces existe $f\in\mpartial[1]$ creciente tal que $\mran{f} = A$.
\end{theorem}

\subsection{CodificaciĂłn de listas}

\begin{definition}[CodificaciĂłn de listas]
    Para cada $n\in\omath[+]$, la \emph{codificaciĂłn de listas} (de longitud $n$) es la funciĂłn $f\in\mfunc[n]$ tal que para cada $x =\pa{x_{1},\dotsc, x_{n}}\in\omath[n]$:
    \[
        f(x) = \prod_{k = 1}^{n}p_{k}^{x_{k} + 1}
    \]
    donde $p_{k}$ es el $k$-Ă©simo nĂşmero primo.
    
    Para cada $n\in\omath[+]$ y cada $x\in\omath[+]$, la codificaciĂłn de $x$ se denotarĂˇ por $\mlist{x}$.

    Para cada $n\in\omath[+]$, define el conjunto de cĂłdigos de longitud $n$ como el conjunto:
    \[
        \mcod[n]\coloneqq\set{\mlist{x}\in\omath: x\in\omath[n]}
    \]

    Se define el conjunto de cĂłdigos como:
    \[
        \mcod\coloneqq\set{1}\cup\bigcup_{n\in\omath[+]}\mcod[n]
    \]
 \end{definition}

\begin{theorem}
     $\mcod$ es primitivo recursivo.
\end{theorem}

\begin{definition}[Longitud de lista]
    Se define la funciĂłn longitud $\mlen\in\mfunc[1]$ para cada $x\in\omath$ como:
    \[
        \mlen\pa{x} =\begin{dcases}
            \mu n\exists y\pa{\pa{y\in\omath[n]}\land\pa{x = \mlist{y}}}, & x\in\mcod\\
            0, & \text{en otro caso}
        \end{dcases}
    \]
\end{definition}

\begin{theorem}
    $\mlen$ es primitiva recursiva.
\end{theorem}

\subsection{MĂˇquinas URM}

\begin{definition}[Instrucciones bĂˇsicas]
    Se define al conjunto de \emph{instrucciones bĂˇsicas} como el siguiente conjunto de sĂ­mbolos:
    \begin{align*}
        \minstr\coloneqq&\set{Z\pa{n}: n\in\omath[+]}\cup\set{S\pa{n}: n\in\omath[+]}\cup\set{T\pa{n, m}: n, m\in\omath[+]}\\
        &\cup\set{J\pa{n, m, k}: n, m, k\in\omath[+]}
    \end{align*}
\end{definition}

\begin{definition}[MĂˇquinas URM]
    Se define al conjunto de \emph{mĂˇquinas URM} (mĂˇquina sin lĂ­mite de registros) como el conjunto de sucesiones finitas de $\minstr$:
    \[
        \murm\coloneqq\bigcup_{n\in\omath[+]}\minstr^{n}
    \]
\end{definition}

\begin{definition}[EjecuciĂłn de mĂˇquinas URM]
    Sean $p, m\in\omath[+]$, $M =\pa{I_{1},\dotsc, I_{p}}\in\murm$ y $x =\pa{x_{1},\dotsc, x_{m}}\in\omath[m]$. La \emph{ejecuciĂłn} de la mĂˇquina $M$ con entrada $x$ es la sucesiĂłn de funciones $\set{f_{n}}_{n\in\omath}\subseteq\mfunc[1]$ tal que para cada $n, k\in\omath$:
    \begin{itemize}
        \item si $n = 0$, entonces:
        \[
            f_{0}\pa{k} =\begin{cases}
                1, & k = 0\\
                x_{k}, & 1\leq k\leq m\\
                0, &\text{en otro caso}
            \end{cases}
        \]
        \item si $f_{n}\pa{0}\leq p$ e $I_{f_{n}\pa{0}} = Z\pa{i}$ para algĂşn $i\in\omath[+]$, entonces:
        \[
            f_{n + 1}\pa{k} =\begin{cases}
                f_{n}\pa{0} + 1, & k = 0\\
                0, & k = i\\
                f_{n}\pa{k}, & \text{en otro caso}
            \end{cases}
        \]
        \item si $f_{n}\pa{0}\leq p$ e $I_{f_{n}\pa{0}} = S\pa{i}$ para algĂşn $i\in\omath[+]$, entonces:
        \[
            f_{n + 1}\pa{k} =\begin{cases}
                f_{n}\pa{k} + 1, & k\in\set{0, i}\\
                f_{n}\pa{k}, & \text{en otro caso}
            \end{cases}
        \]
        \item si $f_{n}\pa{0}\leq p$ e $I_{f_{n}\pa{0}} = T\pa{i, j}$ para algunos $i, j\in\omath[+]$, entonces:
        \[
            f_{n + 1}\pa{k} =\begin{cases}
                f_{n}\pa{0} + 1, & k = 0\\
                f_{n}\pa{i}, & k = j\\
                f_{n}\pa{k}, & \text{en otro caso}
            \end{cases}
        \]
        \item si $f_{n}\pa{0}\leq p$ e $I_{f_{n}\pa{0}} = J\pa{i, j, q}$ para algunos $i, j, q\in\omath[+]$, entonces:
        \[
            f_{n + 1}\pa{k} =\begin{cases}
                q, & k = 0, f_{n}\pa{i} = f_{n}\pa{j}\\
                f_{n}\pa{k}, & \text{en otro caso}
            \end{cases}
        \]
        \item si $f_{n}\pa{0} > p$, entonces $f_{n + 1}\pa{k} = f_{n}\pa{k}$.
    \end{itemize}
    Decimos que la ejecuciĂłn \emph{se detiene} si existe $n\in\omath[+]$ tal que $f_{n}\pa{0} > p$.
\end{definition}

\begin{definition}[Funciones URM-computables]
    Sean $n\in\omath[+]$ y $M\in\murm$. La funciĂłn $\varphi_{M}^{(n)}\in\mfunc[n][\partial]$ asociada a $M$ de aridad $n$ estĂˇ definida para cada $x\in\omath[n]$ como:
    \[
        \varphi_{M}^{(n)}\pa{x} =\begin{dcases}
            f_{k}\pa{1}, & \parbox{0.7\linewidth}{la ejecuciĂłn $\set{f_{n}}_{n\in\omath}\subseteq\mfunc[1]$ de $M$ con entrada $x$ se detiene, con $k$ el mĂ­nimo Ă­ndice donde la ejecuciĂłn se detiene}\\
            \partial, & \text{en otro caso}
        \end{dcases}
    \]

    Sean $n\in\omath[+]$ y $f\in\mfunc[n][\partial]$. $f$ se denomina \emph{URM-computable} si existe $M\in\murm$ tal que $f =\varphi_{M}^{(n)}$.

    Se denota al conjunto de funciones URM-computables como $\murmcomp$.

    Para cada $n\in\omath[+]$ se define el conjunto:
    \[
        \murmcomp[n]\coloneqq\set{f\in\murmcomp: \maridad{f} = n}
    \]
\end{definition}

\begin{theorem}
    $\mpartial\subseteq\murmcomp$
\end{theorem}

\begin{definition}[CodificaciĂłn de mĂˇquinas URM]
    Sea $f:\minstr\to\omath$ definida como sigue:
    \begin{itemize}
        \item para cada $n\in\omath[+]$, $f\pa{Z\pa{n}} =\mlist{0, n}$.
        \item para cada $n\in\omath[+]$, $f\pa{S\pa{n}} =\mlist{1, n}$.
        \item para cada $n, m\in\omath[+]$, $f\pa{T\pa{n, m}} =\mlist{2, n, m}$.
        \item para cada $n, m, k\in\omath[+]$, $f\pa{J\pa{n, m, k}} =\mlist{3, n, m, k}$.
    \end{itemize}
    Se define la codificaciĂłn de $\murm$ como la funciĂłn $f^{*}:\murm\to\omath$ tal que para cada $n\in\omath[+]$ y $M =\pa{I_{1},\dotsc, I_{n}}\in\murm$, $f^{*}\pa{M} =\mlist{0, f\pa{I_{1}},\dotsc, f\pa{I_{n}}}$.

    Definimos al conjunto de cĂłdigos de $\murm$ como $\murmcod\coloneqq\mran{f^{*}}$.
\end{definition}

\begin{theorem}
    $\murmcod$ es primitivo recursivo.
\end{theorem}
