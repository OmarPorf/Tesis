\begin{definition}[$\rca$]
	Se define al subsistema de $\LA$ de \emph{axiomas de comprensiﾃｳn recursiva}, o $\rca$, como el conjunto de axiomas:
	\[
		\rca\coloneqq\mpa\cup\mind{\Sigma_{1}^{0}}\cup\mcomp{\Delta_{1}^{0}}
	\]
\end{definition}

\begin{theorem}
	Sea $S\subseteq\mpow{\omega}$ una $\omega$-estructura. $S$ es un $\omega$-modelo de $\rca$ si y solo si es un ideal de Turing.
\end{theorem}

\begin{proof}
	Para la necesidad debemos demostrar lo siguiente:
	\begin{enumerate}
		\item $S\neq\varnothing$
		
		Consideremos la fﾃｳrmula $\varphi$ dada por $\neg\left(n = n\right)$. Notemos que $\varphi\in\Delta_{0}^{0}$ y luego $\varphi$ es $\Sigma_{1}^{0}$ y $\Pi_{1}^{0}$. Por lo tanto, por $\mcomp{\Delta_{1}^{0}}$ sabemos que existe el conjunto $X\in S$ definido por $\varphi$. Se tiene entonces lo siguiente:
		\[
			X =\left\{n: \varphi\left(n\right)\right\} =\left\{n:\neg\left(n = n\right)\right\} =\varnothing
		\]
		
		Por lo tanto, $\varnothing\in S$, y luego $S\neq\varnothing$.
		
		\item Para cada $X, Y\in S$ se satisface que $X\oplus Y\in S$.
		
		Consideremos a la fﾃｳrmula $\varphi$ dada por:
		\[
			\left(\exists m < n + 1\right)\left(\left(\left(m\in X\right)\land\left(n =\left(m + m\right)\right)\right)\vee\left(\left(m\in Y\right)\land\left(n =\left(\left(m + m\right) + 1\right)\right)\right)\right)
		\]
		Notemos que $\varphi\in\Delta_{0}^{0}$, y entonces por $\mcomp{\Delta_{1}^{0}}$ existe $Z\in S$ definido por $\varphi$. Se obtiene entonces lo siguiente:
		
		\begin{align*}
			Z =&\left\{n: \varphi\left(n\right)\right\}\\
			=&\left\{n: \left(\exists m < n + 1\right)\left(\left(\left(m\in X\right)\land\left(n =\left(m + m\right)\right)\right)\right. \right. \\
			&\left.\left. \vee\left(\left(m\in Y\right)\land\left(n =\left(\left(m + m\right) + 1\right)\right)\right)\right)\right\}\\
			=&\left\{n: \left(\left(\exists m < n + 1\right)\left(\left(m\in X\right)\land\left(n =\left(m + m\right)\right)\right)\right.\right. \\
			&\left.\left.\vee\left(\exists m < n + 1\right) \left(\left(m\in Y\right)\land\left(n =\left(\left(m + m\right) + 1\right)\right)\right)\right) \right\}\\
			=&\left\{n: \left(\exists m < n + 1\right)\left(\left(m\in X\right)\land\left(n =\left(m + m\right)\right)\right)\right\} \\
			&\cup\left\{n: \left(\exists m < n + 1\right) \left(\left(m\in Y\right)\land\left(n =\left(\left(m + m\right) + 1\right)\right)\right)\right\}\\
			=&\left\{2m: m\in X\right\}\cup\left\{2m + 1: m\in Y\right\}\\
			=& X\oplus Y
		\end{align*}
		
		Por lo tanto, $X\oplus Y\in S$.
		
		\item para cada $X, Y\in\mpow{\omega}$, si $X\leq_{T} Y$ y $Y\in S$, entonces $X\in S$.
		
		Ya que $X\leq_{T} Y$, entonces por el teorema de Post (\ref{theo: teorema_de_post}) sabemos que existe una fﾃｳrmula $\varphi$ con parﾃ｡metro $Y$ la cual es $\Sigma_{1}^{0}$ y $\Pi_{1}^{0}$ y que define a $X$. Puesto que $Y\in S$, entonces por $\mcomp{\Delta_{1}^{0}}$ se tiene que existe $Z\in S$ definida por $\varphi$. Por lo tanto, $X = Z\in S$.
	\end{enumerate}
	
	Probemos ahora la suficiencia. Observemos que puesto que $S$ es una $\omega$-estructura, entonces se satisfacen directamente $\mpa$ y $\mind{\Sigma_{1}^{0}}$. Luego, es ﾃｺnicamente necesario demostrar que se satisface el esquema $\mcomp{\Delta_{1}^{0}}$.
	
	Sea $\varphi\left(n\right)\in\Delta_{1}^{0}$ con $\mathrm{Free}\left[\varphi\right] =\left\{n\right\}$, entonces demostremos que existe $X\in S$ definido por $\varphi$. Debido a que $S\neq\varnothing$, podemos suponer que $\varphi\left(n, Y_{1},\dotsc, Y_{k}\right)$ para algunos parﾃ｡metros $Y_{1},\dotsc, Y_{k}\in S$ y $k\in\omega^{+}$. Consideremos al conjunto $X\in\mpow{\omega}$ definido por $\varphi$, entonces por el teorema de Post generalizado (\ref{theo: teorema_de_post_generalizado}) sabemos que $X\leq_{T}Y_{1}\oplus\dotsb\oplus Y_{k}$. Ya que $Y_{1},\dotsc, Y_{k}\in S$, entonces $Y_{1}\oplus\dotsb\oplus Y_{k}\in S$, y por lo tanto, $X\in S$.
\end{proof}

\begin{definition}[Mﾃｭnimo]

\begin{theorem}[Principio del buen orden]
	Todo conjunto $X$ no vacﾃｭo, es decir tal que se satisface $\exists n\left(n\in X\right)$, tiene un elemento mﾃｭnimo.
\end{theorem}
\begin{proof}
	Demostremos por contradicciﾃｳn. Consideremos a $X$ un conjunto no vacﾃｭo, es decir tal que se satisface $\exists n\left(n\in X\right)$, para el cual se cumple lo siguiente:
	\begin{equation}
		\label{eq: buen_orden}
		\forall k\exists m\left(\left(k\notin X\right)\vee\left(\left(m\in X\right)\land\left(m < k\right)\right)\right)
	\end{equation}
	Sea $\varphi\left(p\right)$ la fﾃｳrmula dada por $\left(\forall k\leq p\right)\left(k\notin X\right)$, la cual es $\Delta_{0}^{0}$. Mostremos por $\mind{\Sigma_{1}^{0}}$ que se satisface la fﾃｳrmula $\forall p\varphi\left(p\right)$.
	
	\begin{itemize}
		\item (Base) Si $p = 0$, entonces debemos mostrar que $\left(0\notin X\right)$. Supongamos por el contrario que $\left(0\in X\right)$, entonces por la fﾃｳrmula \eqref{eq: buen_orden} se sigue que existe $m$ tal que $\left(m < 0\right)$ y $\left(m\in X\right)$. Puesto que por los axiomas $\mpa$ se tiene que para cada $n$ se satisface que $\neg\left(n < 0\right)$, entonces llegamos a una contradicciﾃｳn. Por lo tanto, necesariamente $0\notin X$.
		
		\item (Paso inductivo) Sea $n$ tal que se satisface $\varphi\left(n\right)$, entonces demostremos que se satisface $\varphi\left(n + 1\right)$. 
		
		Supongamos que $\left(\left(n + 1\right)\in X\right)$, entonces por la fﾃｳrmula \eqref{eq: buen_orden} se sigue que existe $m$ tal que $\left(m\in X\right)$ y $m <\left(n + 1\right)$, es decir, $\left(m\leq n\right)$. Ya que se satisface $\varphi\left(n\right)$ por hipﾃｳtesis, se tiene entonces que $\left(m\notin X\right)$, lo cual contradice el hecho que $\left(m\in X\right)$ como se dedujo anteriormente. Por lo tanto, necesariamente $\left(\left(n + 1\right)\notin X\right)$.
	\end{itemize}
	
	Se sigue entonces por $\mind{\Sigma_{1}^{0}}$ que:
	\[
		\forall p\varphi\left(p\right)\equiv\forall p\left(\forall k\leq p\right)\left(k\notin X\right)\equiv\forall p\left(p\notin X\right)
	\]
	lo cual contradice el hecho que $X$ era no vacﾃｭo.
	
	En conclusiﾃｳn, por contradicciﾃｳn, se prueba el teorema.
\end{proof}

	\[
		\exists n\forall m\left(\left(m\in X\right)\to\left(m\leq n\right)\right)

\begin{definition}[Mﾃ｡ximo]
	Sea $X$ un conjunto. Se dice que $X$ tiene un \emph{elemento mﾃ｡ximo} si se satisface la siguiente fﾃｳrmula:
	\[
		\exists k\forall m\left(\left(k\in X\right)\land\left(\left(m\in X\right)\to\left(m\leq k\right)\right)\right)
	\]

\begin{theorem}[Principio del mﾃ｡ximo]
	Todo conjunto $X$ no vacﾃｭo y acotado superiormente tiene un elemento mﾃ｡ximo.
\end{theorem}

\begin{proof}
	Defﾃｭnase la fﾃｳrmula $\varphi\left(n\right)$ dada por:
	\begin{equation}
		\label{eq: principio_maximo}
		\left(\left(\forall k\leq n\right)\left(k\notin X\right)\vee\left(\exists p\leq n\right)\left(\forall k\leq n\right)\left(\left(k\in X\right)\to\left(k\leq p\right)\right)\right)
	\end{equation}
	la cual es $\Delta_{0}^{0}$. Observemos que esta fﾃｳrmula se interpreta como que dado $n$ se satisface que todos los elementos menores a $n$ no pertenecen a $X$ o el conjunto $X\cap\set{0,\dotsc, n}$ tiene mﾃ｡ximo. Probemos entonces por $\mind{\Sigma_{1}^{0}}$ que se satisface la fﾃｳrmula $\forall n\varphi\left(n\right)$.
	
	\begin{enumerate}
		\item (Base) Para $n = 0$ se tienen dos casos: $\left(0\in X\right)$ o $\left(0\notin X\right)$. Si se cumple $\left(0\notin X\right)$, entonces directamente se satisface la primera parte de la fﾃｳrmula \eqref{eq: principio_maximo}. Si se satisface $\left(0\in X\right)$, entonces tomando $p = 0$ se satisface la segunda parte de la fﾃｳrmula \eqref{eq: principio_maximo}. En cualquier caso obtenemos que se satisface $\varphi\left(0\right)$.
		
		\item (Paso inductivo) Sea $n$ tal que $\varphi\left(n\right)$. Entonces demostremos que se satisface que $\varphi\left(n + 1\right)$.
		
		Ya que se satisface $\varphi\left(n\right)$, entonces tenemos tres casos: 
		\begin{itemize}
			\item Si se satisface $\left(\left(n + 1\right)\in X\right)$, entonces se satisface la segunda parte de la fﾃｳrmula $\varphi\left(n + 1\right)$ tomando $p = \left(n + 1\right)$.
			
			\item Si se satisface $\left(\left(n + 1\right)\notin X\right)$ y para cada $k$ con $\left(k\leq n\right)$ se cumple que $\left(k\notin X\right)$, entonces claramente se cumple la primer parte de la fﾃｳrmula $\varphi\left(n + 1\right)$.
			
			\item Si se satisface $\left(\left(n + 1\right)\notin X\right)$ y existe un elemento mﾃ｡ximo de los elementos de $X$ acotados por $n$, entonces directamente se satisface la segunda parte de la fﾃｳrmula $\varphi\left(n + 1\right)$.
		\end{itemize}
		Se sigue entonces que en cualquier caso se satisface $\varphi\left(n + 1\right)$. Luego, por $\mind{\Sigma_{1}^{0}}$ se cumple la fﾃｳrmula $\forall n\varphi\left(n\right)$.
		
		Ya que $X$ es acotado superiormente, se deduce que existe $m$ tal que para cada $k$, si $k\in X$, entonces $k\leq m$. Por lo tanto, se satisface particularmente $\varphi\left(m\right)$. 
		
		Ahora, puesto que $X$ es no vacﾃｭo, entonces no se satisface la primera parte de $\varphi\left(m\right)$, de lo cual se sigue que existe $p$ con $\left(p\leq m\right)$ tal que es mﾃ｡ximo de los elementos de $X$ acotados por $m$. Como $m$ ya es cota superior de $X$, entonces tal $p$ es un mﾃ｡ximo de $X$. 
	\end{enumerate}
\end{proof}

\subsection{Los nﾃｺmeros enteros}
En esta secciﾃｳn desarrollaremos la construcciﾃｳn de los nﾃｺmeros enteros, racionales y reales que son necesarios para poder abordar el estudio de los teoremas del Anﾃ｡lisis Real que analizaremos en cada uno de los subsistemas. 

Fuera de $\rca$, recordemos que la caracterﾃｭstica mﾃ｡s importante de los nﾃｺmeros enteros es que existe la resta, y que para cada nﾃｺmero entero $m$ existen dos nﾃｺmeros naturales $n, k$ tales que $m = n - k$. Con esta propiedad podemos codificar cada nﾃｺmero entero con tal par de nﾃｺmeros naturales que se interpretan de la forma anterior. Por ejemplo, si tenemos el par $\pa*{1, 2}$, entonces lo interpretamos como el nﾃｺmero entero $1 - 2 = -1$. Sin embargo, notemos que existen varios pares que nos darﾃｭan el mismo nﾃｺmero entero en ese sentido. Para el caso del $-1$ se tiene que los pares $\pa*{1, 2}$, $\pa*{2, 3}$, $\pa*{3, 4}$ y $\pa*{4, 5}$ se interpretarﾃｭan de la misma forma. En ese caso, decimos que todos estos pares son \emph{equivalentes} para representar al nﾃｺmero $-1$. En general, para cada par de pares $\pa*{n, m}$ y $\pa*{p, q}$, estos son equivalentes si se interpretan como el mismo nﾃｺmero entero, es decir, si $n - m = p - q$, o equivalentemente $n + q = p + m$. Con esta condiciﾃｳn podemos luego definir una relaciﾃｳn de equivalencia que nos codifique la \emph{igualdad} entre nﾃｺmeros enteros en $\rca$. 

\begin{definition}[Igualdad en $\zmath$]
	Se define la relaciﾃｳn $=_{\zmath}\subseteq\nmath[2]\times\nmath[2]$ como sigue para cada $n, m, p, q\in\nmath$:
	\[
		\mlist{n, m} =_{\zmath}\mlist{p, q}\iff n + q = p + m
	\]
\end{definition}

Notemos que esta relaciﾃｳn puede definirse por medio de una fﾃｳrmula $\Delta_{0}^{0}$, por lo que existe debido a $\mcomp{\Delta_{1}^{0}}$.

Ya que vimos que existen elementos diferentes que son \emph{iguales} mediante esta relaciﾃｳn, entonces es necesario demostrar que esta relaciﾃｳn es de equivalencia para que toda la construcciﾃｳn sea coherente.

\begin{theorem}
	\label{theo: relacion_equivalencia_z}
	La relaciﾃｳn $=_{\zmath}$ es de equivalencia.
\end{theorem}
\begin{proof}
	Demostremos entonces que $=_{\zmath}$ es reflexiva, simﾃｩtrica y transitiva. Sean $n, m, p, q, r, s\in\nmath$.
	\begin{enumerate}
		\item Reflexividad
		
		Puesto que $n + m = m + n$, entonces se sigue directamente $\mlist{n, m} =_{\zmath}\mlist{n, m}$. Por lo tanto, $=_{\zmath}$ es reflexiva.
		
		\item Simetrﾃｭa
		
		Supongamos que $\mlist{n, m} =_{\zmath}\mlist{p, q}$, entonces por definiciﾃｳn se satisface que $n + q = p + m$, o equivalentemente, $p + m = n + q$. De lo anterior se sigue que $\mlist{p, q} =_{\zmath}\mlist{n, m}$. Por lo tanto, $=_{\zmath}$ es simﾃｩtrica.
		
		\item Transitividad
		
		Supongamos que $\mlist{n, m} =_{\zmath}\mlist{p, q}$ y $\mlist{p, q} =_{\zmath}\mlist{r, s}$, entonces se satisface que $n + q = p + m$ y $p + s = r + q$. Sumando ambas igualdades obtenemos lo siguiente:
		\[
			\begin{NiceArray}{c}
				\pa*{n + q} +\pa*{p + s} =\pa*{p + m} +\pa*{r + q}\\
				\pa*{n + s} +\pa*{p + q} =\pa*{r + m} +\pa*{p + q}\\
				n + s = r + m
			\end{NiceArray}
		\]
		Lo cual implica que $\mlist{n, m} =_{\zmath}\mlist{r, s}$. Por lo tanto, $=_{\zmath}$ es transitiva
	\end{enumerate}
	En conclusiﾃｳn, la relaciﾃｳn $=_{\zmath}$ es de equivalencia
\end{proof}

Mﾃ｡s aﾃｺn, puesto que $=_{\zmath}$ es un conjunto que existe en $\rca$, entonces la clase de equivalencia de cualquier par de naturales existe ya que puede definirse fﾃ｡cilmente por una fﾃｳrmula $\Delta_{0}^{0}$.

Ya que tenemos el concepto de igualdad en los enteros, podemos definir lo que es la suma y producto en ellos. Fuera de $\rca$, consideremos dos nﾃｺmeros enteros $x = n - m$ y $y = p - q$, entonces su suma y producto satisface lo siguiente:
\[
	\begin{NiceArray}{c}
		x + y =\pa*{n - m} +\pa*{p - q} =\pa*{n + p} -\pa*{m + q}\\
		xy =\pa*{n - m}\pa*{p - q} = np - nq - mp + mq =\pa*{np + mq} -\pa*{nq + mp}
	\end{NiceArray}
\]

De esta forma, podrﾃｭamos definir la suma y producto de enteros como los pares $\pa*{n + p, m + q}$ y $\pa*{np + mq, nq + mp}$ respectivamente. Con esto definimos sus equivalentes en $\rca$.

\begin{definition}
	\label{def: suma_producto_z_aux}
	Definimos las siguientes funciones:
	\[
		\begin{NiceArray}{cc}
			f:\nmath[2]\times\nmath[2]\to\nmath & g:\nmath[2]\times\nmath[2]\to\nmath\\
			\pa*{\mlist{n, m},\mlist{p, q}}\mapsto\mlist{n + p, m + q} &\pa*{\mlist{n, m},\mlist{p, q}}\mapsto\mlist{np + mq, nq + mp}
		\end{NiceArray}
	\]
\end{definition}

Por el momento no les denominamos suma ni producto debido a que necesitaremos redefinirlas cuando establezcamos propiamente el conjunto de nﾃｺmeros enteros. 

En cambio, demostremos que estas dos operaciones se ``comportan bien'', es decir, que son independientes del representante, son conmutativas, asociativas y distributivas.

\begin{lemma}
	\label{lemma: independencia_representantes_z}
	Para cada $x, y, z, w\in\nmath[2]$ se satisface que si $x =_{\zmath} z$ y $y =_{\zmath} w$, entonces $f\pa*{x, y} =_{\zmath} f\pa*{z, w}$ y $g\pa*{x, y} =_{\zmath} g\pa*{z, w}$.
\end{lemma}

\begin{proof}
	Sean $n, m, p, q, r, s, t, u\in\nmath$ tales que $x =\mlist{n, m}$, $y =\mlist{p, q}$, $z =\mlist{r, s}$ y $w =\mlist{t, u}$.
	\begin{enumerate}
		\item Mostremos que $f\pa*{x, y} =_{\zmath} f\pa*{z, w}$.
		
		Puesto que $x =_{\zmath} z$ y $y =_{\zmath} w$, entonces $n + s = r + m$ y $p + u = t + q$. Luego:
		\begin{align*}
			\pa*{n + s} +\pa*{p + u} &=\pa*{r + m} +\pa*{t + q}\\
			\pa*{n + p} +\pa*{s + u} &=\pa*{r + t} +\pa*{m + q}\\
			\mlist{n + p, m + q} &=_{\zmath}\mlist{r + t, s + u}\\
			f\pa*{x, y} &=_{\zmath} f\pa*{z, w}
		\end{align*}
		
		\item Mostremos que $g\pa*{x, y} =_{\zmath} g\pa*{z, w}$.
		
		Puesto que $x =_{\zmath} z$ y $y =_{\zmath} w$, entonces $n + s = r + m$ y $p + u = t + q$. Luego:
		\begin{align*}
			p + u &= t + q\\
			n\pa*{p + u} &= n\pa*{t + q}\\
			np + nu &= nt + nq\\
			np + nu + mq &= nt + nq + mq\\
			np + mq + nu &= nq + mq + nt\\
			np + mq + nu + su &= nq + mq + nt + su\\
			np + mq + \pa*{n + s}u &= nq + mq + nt + su\\
			np + mq + \pa*{r + m}u &= nq + mq + nt + su\\
			np + mq + ru + mu &= nq + mq + nt + su\\
			np + mq + ru + mu + mt &= nq + mq + mt + nt + su\\
			np + mq + ru + mu + mt &= nq + m\pa*{t + q} + nt + su\\
			np + mq + ru + mu + mt &= nq + m\pa*{p + u} + nt + su\\
			np + mq + ru + mu + mt &= nq + mp + mu + nt + su\\
			np + mq + ru + mt &= nq + mp + nt + su\\
			np + mq + ru + mt + st &= nq + mp + nt + st + su\\
			np + mq + ru + mt + st &= nq + mp + \pa*{n + s}t + su\\
			np + mq + ru + mt + st &= nq + mp + \pa*{r + m}t + su\\
			np + mq + ru + mt + st &= nq + mp + rt + mt + su\\
			np + mq + ru + st &= nq + mp + rt + su\\
			\pa*{np + mq} + \pa*{ru + st} &= \pa*{nq + mp} + \pa*{rt + su}\\
			\mlist{np + mq, nq + mp} &=_{\zmath}\mlist{rt + su, ru + st}\\
			g\pa*{x, y} &=_{\zmath} g\pa*{z, w}
		\end{align*}
	\end{enumerate}
\end{proof}
\begin{lemma}
	\label{lemma: prop_suma_prod_z}
	Para cada $x, y, z\in\nmath[2]$ se satisface lo siguiente:
	\begin{enumerate}
		\item $f\pa*{x, y} = f\pa*{y, x}$
		\item $g\pa*{x, y} = g\pa*{y, x}$
		\item $f\pa*{x, f\pa*{y, z}} = f\pa*{f\pa*{x, y}, z}$
		\item $g\pa*{x, g\pa*{y, z}} = g\pa*{g\pa*{x, y}, z}$
		\item $g\pa*{x, f\pa*{y, z}} = f\pa*{g\pa*{x, y}, g\pa*{x, z}}$
	\end{enumerate}
\end{lemma}
\begin{proof}
	Sean $n, m, p, q, r, s\in\nmath$ tales que $x =\mlist{n, m}$, $y =\mlist{p, q}$ y $z =\mlist{r, s}$.
	\begin{enumerate}
		\item $f\pa*{x, y} = f\pa*{y, x}$
		
		Se tiene lo siguiente:
		\[
			f\pa*{x, y} =\mlist{n + p, m + q} =\mlist{p + n, q + m} = f\pa*{y, x}
		\]
		\item $g\pa*{x, y} = g\pa*{y, x}$
		
		Se satisface lo siguiente:
		\[
			g\pa*{x, y} =\mlist{np + mq, nq + mp} =\mlist{pn + qm, pm + qn} = g\pa*{y, x}
		\]
		\item $f\pa*{x, f\pa*{y, z}} = f\pa*{f\pa*{x, y}, z}$
		
		Se satisface lo siguiente:
		\begin{align*}
			f\pa*{x, f\pa*{y, z}} &= f\pa*{x,\mlist{p + r, q + s}}\\
			&=\mlist{n +\pa*{p + r}, m +\pa*{q + s}}\\
			&=\mlist{\pa*{n + p} + r, \pa*{m + q} + s}\\
			&= f\pa*{\mlist{n + p, m + q}, z}\\
			&= f\pa*{f\pa*{x, y}, z}
		\end{align*}
		\item $g\pa*{x, g\pa*{y, z}} = g\pa*{g\pa*{x, y}, z}$
		
		Se satisface lo siguiente:
		\begin{align*}
			g\pa*{x, g\pa*{y, z}} &= g\pa*{x,\mlist{pr + qs, ps + qr}}\\
			&=\mlist{n\pa*{pr + qs} + m\pa*{ps + qr}, n\pa*{ps + qr} + m\pa*{pr + qs}}\\
			&=\mlist{npr + nqs + mps + mqr, nps + nqr + mpr + mqs}\\
			&=\mlist{\pa*{np + mq}r + \pa*{nq + mp}s, \pa*{np + mq}s + \pa*{nq + mp}r}\\
			&= g\pa*{\mlist{np + mq, nq + mp}, z}\\
			&= g\pa*{g\pa*{x, y}, z}
		\end{align*}
		\item $g\pa*{x, f\pa*{y, z}} = f\pa*{g\pa*{x, y}, g\pa*{x, z}}$
		
		Se satisface lo siguiente:
		\begin{align*}
			g\pa*{x, f\pa*{y, z}} &= g\pa*{x,\mlist{p + r, q + s}}\\
			&=\mlist{n\pa*{p + r} + m\pa*{q + s}, n\pa*{q + s} + m\pa*{p + r}}\\
			&=\mlist{np + nr + mq + ms, nq + ns + mp + mr}\\
			&=\mlist{\pa*{np + mq} +\pa*{nr + ms}, \pa*{nq + mp} +\pa*{ns + mr}}\\
			&= f\pa*{\mlist{np + mq, nq + mp}, \mlist{nr + ms, ns + mr}}\\
			&= f\pa*{g\pa*{x, y}, g\pa*{x, z}}
		\end{align*}
	\end{enumerate}
\end{proof}

\begin{lemma}
	\label{lemma: 0_enteros}
	Para cada $n\in\nmath$ se satisface que $\mlist{0, 0} =_{\zmath}\mlist{n, n}$.
\end{lemma}
\begin{proof}
	Se tiene directamente puesto que $n + 0 = 0 + n$.
\end{proof}

Con esto podemos ahora definir propiamente lo que son los nﾃｺmeros enteros en $\rca$. Aunque lo mﾃ｡s usual en teorﾃｭa de conjuntos es usar cada clase de equivalencia como cada nﾃｺmero entero y luego el conjunto de nﾃｺmeros enteros se defina como el conjunto cociente $\nmath[2] / =_{\zmath}$, en $\rca$ no existen los conjuntos de conjuntos de nﾃｺmeros naturales, por lo que no podemos irnos directamente por esa definiciﾃｳn. En cambio, de cada clase de equivalencia podrﾃｭamos elegir un ﾃｺnico elemento como representante y luego los nﾃｺmeros enteros serﾃｭan tal conjunto de representantes, el cual sﾃｭ puede existir ya que es un conjunto de naturales, lo cual sﾃｭ estﾃ｡ permitido. Puesto que ya sabemos que dado un conjunto no vacﾃｭo podemos determinar su elemento mﾃｭnimo, entonces de cada clase utilizaremos su elemento mﾃｭnimo como representante.

\begin{definition}[Nﾃｺmeros enteros]
	Definimos la funciﾃｳn $Z:\nmath[2]\to\nmath$ tal que para cada $x\in\nmath[2]$, $Z\pa*{x} =\min\bra*{x}_{=_{\zmath}}$. Se define al conjunto de \emph{nﾃｺmeros enteros} como el conjunto $\zmath\subseteq\nmath$ tal que:
	\[
		\zmath\coloneqq\mran{Z}
	\]
	es decir, un \emph{nﾃｺmero entero} es cualquier natural $x\in\nmath$ tal que existe $n\in\nmath[2]$ para el cual $x =\min\bra*{n}_{=_{\zmath}}$.
\end{definition}

Notemos que $\zmath$ existe por $\mcomp{\Delta_{1}^{0}}$. Ademﾃ｡s, $Z$ estﾃ｡ bien definida porque para cada $x\in\nmath[2]$ se satisface que $x\in\bra*{x}_{=_{\zmath}}$ y luego su mﾃｭnimo existe. Otra observaciﾃｳn es que claramente para cada $x, y\in\nmath[2]$ tales que $x =_{\zmath} y$ se satisface que $Z\pa*{x} = Z\pa*{y}$, pues $x$ y $y$ forman parte de la misma clase y luego tienen el mismo mﾃｭnimo.

Ahora podemos tambiﾃｩn definir adecuadamente la suma y producto de $\zmath$, con lo que podemos demostrar que $\zmath$ es un anillo con unidad.

\begin{definition}[Suma y producto en $\zmath$]
	Se define la \emph{suma} y \emph{producto} en $\zmath$ como las siguientes funciones:
	\[
		\begin{NiceArray}{cc}
			+_{\zmath}:\zmath[2]\to\zmath & \cdot_{\zmath}:\zmath[2]\to\zmath\\
			\pa*{x, y}\mapsto Z\pa*{f\pa*{x, y}} &\pa*{x, y}\mapsto Z\pa*{g\pa*{x, y}}
		\end{NiceArray}
	\]
	donde $f$ y $g$ son las funciones de la definiciﾃｳn \ref{def: suma_producto_z_aux}.
\end{definition}

\begin{theorem}
	\label{theo: enteros_anillo}
	El conjunto de enteros $\zmath$ es un anillo conmutativo con unidad con las operaciones $+_{\zmath}$ y $\cdot_{\zmath}$.
\end{theorem}

\begin{proof}
	Sean $x, y, z\in\zmath$, entonces mostremos cada una de las siguientes propiedades con ayuda de los lemas \ref{lemma: independencia_representantes_z}, \ref{lemma: prop_suma_prod_z} y \ref{lemma: 0_enteros}:
	\begin{enumerate}
		\item Conmutatividad para la suma
		\[
			x +_{\zmath} y = Z\pa*{f\pa*{x, y}} = Z\pa*{f\pa*{y, x}} = y +_{\zmath} x
		\]
		\item Asociatividad para la suma
		\begin{align*}
			x +_{\zmath}\pa*{y +_{\zmath} z} &= Z\pa*{f\pa*{x, Z\pa*{f\pa*{y, z}}}}\\
			&= Z\pa*{f\pa*{x, f\pa*{y, z}}}\\
			&= Z\pa*{f\pa*{f\pa*{x, y}, z}}\\
			&= Z\pa*{f\pa*{Z\pa*{f\pa*{x, y}}, z}}\\
			&=\pa*{x +_{\zmath} y} +_{\zmath} z
		\end{align*}
		\item Existencia del neutro aditivo
		
		Definamos al entero $0_{\zmath}\coloneqq Z\pa*{\mlist{0, 0}}$, entonces si $x =\mlist{n, m}$ para algunos $n, m\in\nmath$, se tiene que:
		\begin{align*}
			x +_{\zmath} 0_{\zmath} &= Z\pa*{f\pa*{x, 0_{\zmath}}} \\
			&= Z\pa*{f\pa*{\mlist{n, m},Z\pa*{\mlist{0, 0}}}} \\
			&= Z\pa*{f\pa*{\mlist{n, m}, \mlist{0, 0}}}\\
			&= Z\pa*{\mlist{n, m}}\\
			&= Z\pa*{x} = x
		\end{align*}
		\item Existencia de inversos aditivos
		
		Si $x =\mlist{n, m}$ para algunos $n, m\in\nmath$, entonces definamos $-x\coloneqq Z\pa*{\mlist{m, n}}$. Luego:
		\begin{align*}
			x +_{\zmath}\pa*{-x} &= Z\pa*{f\pa*{x, -x}}\\
			&= Z\pa*{f\pa*{\mlist{n, m}, Z\pa*{\mlist{m, n}}}}\\
			&= Z\pa*{f\pa*{\mlist{n, m},\mlist{m, n}}}\\
			&= Z\pa*{\mlist{n + m, n + m}}\\
			&= Z\pa*{\mlist{0, 0}} = 0_{\zmath}
		\end{align*}
		\item Conmutatividad para el producto
		\[
		x \cdot_{\zmath} y = Z\pa*{g\pa*{x, y}} = Z\pa*{g\pa*{y, x}} = y \cdot_{\zmath} x
		\]
		\item Asociatividad para el producto
		\begin{align*}
			x \cdot_{\zmath}\pa*{y \cdot_{\zmath} z} &= Z\pa*{g\pa*{x, Z\pa*{g\pa*{y, z}}}}\\
			&= Z\pa*{g\pa*{x, g\pa*{y, z}}}\\
			&= Z\pa*{g\pa*{g\pa*{x, y}, z}}\\
			&= Z\pa*{g\pa*{Z\pa*{g\pa*{x, y}}, z}}\\
			&=\pa*{x \cdot_{\zmath} y} \cdot_{\zmath} z
		\end{align*}
		\item Existencia del neutro multiplicativo
		
		Definamos al entero $1_{\zmath}\coloneqq Z\pa*{\mlist{1, 0}}$, entonces si $x =\mlist{n, m}$ para algunos $n, m\in\nmath$, se tiene que:
		\begin{align*}
			x \cdot_{\zmath} 1_{\zmath} &= Z\pa*{g\pa*{x, 1_{\zmath}}} \\
			&= Z\pa*{g\pa*{\mlist{n, m},Z\pa*{\mlist{1, 0}}}} \\
			&= Z\pa*{g\pa*{\mlist{n, m}, \mlist{1, 0}}}\\
			&= Z\pa*{\mlist{n\cdot 1 + m\cdot 0, n\cdot 0 + m\cdot 1}}\\
			&= Z\pa*{\mlist{n, m}}\\
			&= Z\pa*{x} = x
		\end{align*}
		
		\item Distributividad
		
		\begin{align*}
			x\cdot_{\zmath}\pa*{y +_{\zmath} z} &= Z\pa*{g\pa*{x, Z\pa*{f\pa*{y, z}}}}\\
			&= Z\pa*{g\pa*{x, f\pa*{y, z}}}\\
			&= Z\pa*{f\pa*{g\pa*{x, y}, g\pa*{x, z}}}\\
			&= Z\pa*{f\pa*{Z\pa*{g\pa*{x, y}}, Z\pa*{g\pa*{x, z}}}}\\
			&=\pa*{x\cdot_{\zmath} y} +_{\zmath}\pa*{x\cdot_{\zmath} z}
		\end{align*}
	\end{enumerate}
	Por lo tanto, $\zmath$ es un anillo conmutativo con unidad con las operaciones $+_{\zmath}$ y $\cdot_{\zmath}$.
\end{proof}

Ahora, para definir a los nﾃｺmeros racionales es fundamental tener el concepto de orden en $\zmath$. Fuera de $\rca$, si tenemos dos nﾃｺmeros enteros $x = n - m$ y $y = p - q$ para algunos naturales $n$, $m$, $p$ y $q$, sabemos que $x < y$ si y solo si $n - m < p - q$, o equivalentemente, $n + q < p + m$. De esta forma podemos tambiﾃｩn extender el concepto de orden a $\zmath$ en $\rca$ como tal relaciﾃｳn.

\begin{definition}[Orden en $\zmath$]
	Se define la \emph{relaciﾃｳn de orden} en $\zmath$ como la relaciﾃｳn $<_{\zmath}\subseteq\zmath[2]$ tal que para cada $x =\mlist{n, m}, y =\mlist{p, q}\in\zmath$:
	\[
		x <_{\zmath} y\iff n + q < p + m
	\]
\end{definition}

Se tiene que $<_{\zmath}$ existe por $\mcomp{\Delta_{1}^{0}}$, y ademﾃ｡s es un orden lineal.

\begin{theorem}
	La relaciﾃｳn de orden $<_{\zmath}$ es un orden lineal en $\zmath$.
\end{theorem}
\begin{proof}
	Sean $x =\mlist{n, m}, y =\mlist{p, q}, z =\mlist{r, s}\in\zmath$, entonces mostremos que $<_{\zmath}$ es irreflexiva, transitiva y tricotﾃｳmica.
	\begin{enumerate}
		\item Irreflexividad
		
		Mostremos que $\neg\pa*{x <_{\zmath} x}$. Puesto que $n + m = m + n$, entonces se satisface que $\neg\pa*{n + m < m + n}$, por lo que $\neg\pa*{x <_{\zmath} x}$.
		
		\item Transitividad
		
		Supongamos que $x <_{\zmath} y$ y $y <_{\zmath} z$, entonces $n + q < p + m$ y $p + s < r + q$, por lo que se tiene lo siguiente:
		\begin{align*}
			\pa*{n + q} +\pa*{p + s} &<\pa*{p + m} +\pa*{r + q}\\
			\pa*{n + s} +\pa*{p + q} &<\pa*{r + m} +\pa*{p + q}\\
			n + s &< r + m
		\end{align*}
		Por lo tanto, $x <_{\zmath} z$.
		
		\item Tricotomﾃｭa
		
		Como $n, m, p, q\in\nmath$, y $\nmath$ es tricotﾃｳmico, entonces se satisface una y solo una de las siguientes afirmaciones: $n + q = p + m$, $n + q < p + m$ o $n + q > p + m$. 
		
		Si $n + q = p + m$, entonces $x =_{\zmath} y$, lo cual implicarﾃｭa que $x = y$ ya que al ser enteros, serﾃｭan representantes de la misma clase de equivalencia.
		
		Si $n + q < p + m$, entonces $x <_{\zmath} y$ por definiciﾃｳn, y si $n + q > p + m$, tambiﾃｩn por definiciﾃｳn $y <_{\zmath} x$.
		
		Por lo tanto, $<_{\zmath}$ es tricotﾃｳmica.
	\end{enumerate}
	Por lo tanto, $<_{\zmath}$ es un orden lineal en $\zmath$.
\end{proof}

Para finalizar la construcciﾃｳn de $\zmath$, mostremos que los nﾃｺmeros naturales se pueden encajar en $\zmath$, lo cual podemos entender como que existe un subconjunto de $\zmath$ que se comporta de manera idﾃｩntica como lo harﾃｭa $\nmath$ en cuanto a la suma, producto y orden. Formalmente eso significa que debe existir una funciﾃｳn inyectiva de $\nmath$ a $\zmath$ que conserve la suma, producto y orden de $\nmath$ a $\zmath$.

Puesto que nosotros ya interpretﾃ｡bamos cada par $\mlist{n, m}$ de naturales como la resta $n - m$, entonces los nﾃｺmeros naturales serﾃｭan los enteros tales que su segunda componente es $0$. Por ejemplo, el nﾃｺmero $2$ en $\zmath$ serﾃｭa el representante de la clase de $\mlist{2, 0}$, pues $2 = 2 - 0$.

Aseguramos que esta traducciﾃｳn de naturales a enteros es aquella funciﾃｳn inyectiva que nos permite encajar los naturales en los enteros.

\begin{theorem}[Encaje de $\nmath$ en $\zmath$]
	Defﾃｭnase la funciﾃｳn $N:\nmath\to\zmath$ tal que para cada $n\in\nmath$, $N\pa*{n}\coloneqq Z\pa*{\mlist{n, 0}}$. Entonces $N$ es inyectiva y satisface las siguientes propiedades:
	\begin{enumerate}
		\item Para cada $n, m\in\nmath$, $N\pa*{n + m} = N\pa*{n} +_{\zmath} N\pa*{m}$.
		\item Para cada $n, m\in\nmath$, $N\pa*{nm} = N\pa*{n} \cdot_{\zmath} N\pa*{m}$.
		\item Para cada $n, m\in\nmath$, $n < m$ si y solo si $N\pa*{n} <_{\zmath} N\pa*{m}$.
	\end{enumerate}
\end{theorem}

\begin{proof}
	Sean $n, m\in\nmath$, entonces:
	\begin{enumerate}
		\item Inyectividad de $N$
		
		Supongamos que $N\pa*{n} = N\pa*{m}$, entonces se tiene que $Z\pa*{\mlist{n, 0}} = Z\pa*{\mlist{m, 0}}$, lo cual significa que $\mlist{n, 0} =_{\zmath}\mlist{m, 0}$, es decir que $n + 0 = 0 + m$, o equivalentemente, $n = m$. Por lo tanto, $N$ es inyectiva.
		
		\item Conservaciﾃｳn de la suma
		
		Se tiene lo siguiente:
		\begin{align*}
			N\pa*{n + m} &= Z\pa*{\mlist{n + m, 0}}\\
			&= Z\pa*{f\pa*{\mlist{n, 0},\mlist{m, 0}}}\\
			&= Z\pa*{f\pa*{Z\pa*{\mlist{n, 0}}, Z\pa*{\mlist{m, 0}}}}\\
			&= Z\pa*{f\pa*{N\pa*{n}, N\pa*{m}}}\\
			&= N\pa*{n} +_{\zmath} N\pa*{m}
		\end{align*}
		
		\item Conservaciﾃｳn del producto
		
		Se tiene lo siguiente:
		\begin{align*}
			N\pa*{nm} &= Z\pa*{\mlist{nm, 0}}\\
			&= Z\pa*{\mlist{nm + 0\cdot 0, n\cdot 0 + 0\cdot m}}\\
			&= Z\pa*{g\pa*{\mlist{n, 0},\mlist{m, 0}}}\\
			&= Z\pa*{g\pa*{Z\pa*{\mlist{n, 0}}, Z\pa*{\mlist{m, 0}}}}\\
			&= Z\pa*{g\pa*{N\pa*{n}, N\pa*{m}}}\\
			&= N\pa*{n} \cdot_{\zmath} N\pa*{m}
		\end{align*}
		
		\item Conservaciﾃｳn del orden
		
		Sean $N\pa*{n} =\mlist{p, q}$ y $N\pa*{m} =\mlist{r, s}$ para algunos $p, q, r, s\in\nmath$, entonces sabemos que $n + q = p$ y $m + s = r$ por definiciﾃｳn de $N$, entonces:
		\begin{align*}
			n < m &\iff n + q < m + q\\
			&\iff p < m + q\\
			&\iff p + s < m + q + s\\
			&\iff p + s < r + q\\
			&\iff \mlist{p, q} <_{\zmath}\mlist{r, s}\\
			&\iff N\pa*{n} <_{\zmath} N\pa*{m}
		\end{align*}
	\end{enumerate}
\end{proof}

Con esto hemos mostrado que al trabajar con los enteros, tambiﾃｩn podemos hacer referencia a los naturales si es que necesitamos tratar con ellos. Aunque cuando hablamos de los naturales en $\zmath$ no necesariamente estamos realizando las mismas operaciones en el propio $\nmath$, es decir, no necesariamente $2_{\zmath} +_{\zmath} 3_{\zmath} = 2 + 3$, la inyecciﾃｳn $N$ nos permite traducir cada expresiﾃｳn de un conjunto a otro de manera sencilla, indicﾃ｡ndonos que podemos ignorar esa diferencia entre $\nmath$ y $\mran{N}$, y en cambio tratarlos como el mismo conjunto. Por eso mismo, no haremos distinciﾃｳn simbﾃｳlica entre $+_{\zmath}$ y la suma en $\nmath$ al trabajar en $\zmath$ debido a ese encaje.

\subsection{Los nﾃｺmeros racionales}

Ya que hemos construido los nﾃｺmeros enteros, ya directamente podemos definir lo que son los nﾃｺmeros racionales, lo cuales son los mﾃ｡s fundamentales para el estudio del Anﾃ｡lisis Real que realizaremos. 

Fuera de $\rca$, los nﾃｺmeros racionales se caracterizan por tener \emph{divisiﾃｳn}, y en general cualquier nﾃｺmero racional $q$ puede expresarse como un cociente entre nﾃｺmeros enteros $q = n / m$, donde $m > 0$. De manera similar podemos codificar un nﾃｺmero racional como un par de nﾃｺmeros enteros interpretados de la forma anterior. Por ejemplo, el nﾃｺmero $2/3$ serﾃｭa el par $(2, 3)$. Sin embargo, el par no es ﾃｺnico, pues $2/3$ tambiﾃｩn puede representarse con los pares $(4, 6)$, $(6, 9)$, $(8, 12)$, etc. Entonces todos estos pares se deberﾃｭan considerar \emph{equivalentes} al interpretarse como el mismo racional. En general, para cada par de pares $(p, q)$ y $(r, s)$ que representan a los racionales $p/q$ y $r/s$, estos son equivalentes si y solo si $p/q = r/s$, o equivalentemente $ps = rq$. De esta manera podemos definir la relaciﾃｳn de equivalencia entre pares de enteros en $\rca$ que codifica la igualdad entre racionales.

\begin{definition}[Igualdad en $\qmath$]
	Se define la relaciﾃｳn $=_{\qmath}\subseteq\pa*{\zmath\times\zmath[+]}\times\pa*{\zmath\times\zmath[+]}$ como sigue para cada $n, m, p, q\in\nmath$:
	\[
	\mlist{n, m} =_{\qmath}\mlist{p, q}\iff nq = pm
	\]
\end{definition}

Aquﾃｭ $\zmath[+]$ se entiende como el conjunto de enteros positivos, es decir el conjunto $\zmath[+] =\set*{n: \pa*{n\in\zmath}\land\pa*{n >_{\zmath} 0}}$. Se tiene tambiﾃｩn que esta relaciﾃｳn existe por $\mcomp{\Delta_{1}^{0}}$. 

De manera anﾃ｡loga al teorema \ref{theo: relacion_equivalencia_z} se demuestra que $=_{\qmath}$ es de equivalencia, por lo que para cada $q\in\zmath\times\zmath[+]$ existe su clase de equivalencia $\bra*{q}_{=_{\qmath}}$ por $\mcomp{\Delta_{1}^{0}}$. Entonces, de manera similar definiremos a los nﾃｺmeros racionales como el conjunto de elementos mﾃｭnimos de cada clase de equivalencia.

\begin{definition}[Nﾃｺmeros racionales]
	Defﾃｭnase la funciﾃｳn $Q:\zmath\times\zmath[+]\to\nmath$ tal que para cada $q\in\zmath\times\zmath[+]$, $Q\pa*{q} =\min\bra*{q}_{=_{\qmath}}$. Entonces definimos al \emph{conjunto de nﾃｺmeros racionales} como el subconjunto $\qmath\subseteq\nmath$ tal que:
	\[
		\qmath\coloneqq\mran{Q}
	\]
	es decir, un \emph{nﾃｺmero racional} es cualquier natural $q\in\nmath$ tal que existe $n\in\zmath\times\zmath[+]$ para el cual $q =\min\bra*{n}_{=_{\qmath}}$.
\end{definition}

De manera anﾃ｡loga podemos definir las operaciones en $\qmath$ por medio de dos funciones auxiliares que codifican la suma y producto de racionales usando los pares de enteros. Fuera de $\rca$, si tenemos dos racionales $x = n/m$ y $y = p/q$, entonces obtenemos lo siguiente:
\[
	\begin{NiceArray}{c}
		x + y =\dfrac{n}{m} +\dfrac{p}{q} =\dfrac{nq + mp}{mq}\\
		xy =\pa*{\dfrac{n}{m}}\pa*{\dfrac{p}{q}} =\dfrac{np}{mq}
	\end{NiceArray}
\]

Por lo que la suma y producto estarﾃｭan codificados como los pares $\pa*{nq + mp, mq}$ y $\pa*{np, mq}$ respectivamente, lo cual podemos llevar a $\rca$ de la siguiente forma.

\begin{definition}
	\label{def: suma_producto_q_aux}
	Definimos las siguientes funciones:
	\[
	\begin{NiceArray}{cc}
		f:\pa*{\zmath\times\zmath[+]}\times\pa*{\zmath\times\zmath[+]}\to\nmath & g:\pa*{\zmath\times\zmath[+]}\times\pa*{\zmath\times\zmath[+]}\to\nmath\\
		\pa*{\mlist{n, m},\mlist{p, q}}\mapsto\mlist{nq + mp, mq} &\pa*{\mlist{n, m},\mlist{p, q}}\mapsto\mlist{np, mq}
	\end{NiceArray}
	\]
\end{definition}

Al igual que en caso de los enteros, es necesario asegurar que estas operaciones son independientes del representante bajo la igualdad en $\qmath$.

\begin{lemma}
	\label{lemma: independencia_representantes_q}
	Para cada $x, y, z, w\in\zmath\times\zmath[+]$ se satisface que si $x =_{\qmath} z$ y $y =_{\qmath} w$, entonces $f\pa*{x, y} =_{\qmath} f\pa*{z, w}$ y $g\pa*{x, y} =_{\qmath} g\pa*{z, w}$.
\end{lemma}

\begin{proof}
	Sean $n, p, r, t\in\zmath$ y $m, q, s, u\in\zmath[+]$ tales que $x =\mlist{n, m}$, $y =\mlist{p, q}$, $z =\mlist{r, s}$ y $w =\mlist{t, u}$ donde ademﾃ｡s $x =_{\qmath} z$ y $y =_{\qmath} w$. Se tiene entonces que $ns = rm$ y $pu = tq$.
	
	\begin{enumerate}
		\item Para $f$ se tiene lo siguiente:
		\begin{align*}
			\pa*{nq + mp}\pa*{su} &= nqsu + mpsu\\
			&= rmqu + tqms\\
			&=\pa*{ru + st}\pa*{mq}
		\end{align*}
		Por lo que $f\pa*{x, y} =_{\qmath} f\pa*{z, w}$.
		\item Para $g$ se tiene lo siguiente:
		\[
			\pa*{np}\pa*{su} =\pa*{ns}\pa*{pu} =\pa*{rm}\pa*{tq}
		\]
		Por lo que $g\pa*{x, y} =_{\qmath} g\pa*{z, w}$.
	\end{enumerate}
\end{proof}

Una observaciﾃｳn a realizar es que en los racionales se cumple la cancelaciﾃｳn en la divisiﾃｳn, es decir, si nosotros tenemos el racional $np/nq$ donde $n\neq 0$, entonces $np/nq = p/q$. Traducido a la construcciﾃｳn que estamos realizando se obtiene el siguiente teorema.

\begin{lemma}
	\label{lemma: cancelacion_cociente}
	Para cada $p\in\zmath$ y cada $n, q\in\zmath[+]$ se satisface que $\mlist{np, nq} =_{\qmath}\mlist{p, q}$.
\end{lemma}

\begin{proof}
	Se sigue directamente puesto que $\pa*{np}q = p\pa*{nq}$.
\end{proof}

Con esto podemos demostrar que estas funciones se comportan bien, lo cual significa que son conmutativas, asociativas y distributivas.

\begin{lemma}
	\label{lemma: prop_suma_prod_q}
	Para cada $x, y, z\in\zmath\times\zmath[+]$ se satisface lo siguiente:
	\begin{enumerate}
		\item $f\pa*{x, y} = f\pa*{y, x}$
		\item $g\pa*{x, y} = g\pa*{y, x}$
		\item $f\pa*{x, f\pa*{y, z}} = f\pa*{f\pa*{x, y}, z}$
		\item $g\pa*{x, g\pa*{y, z}} = g\pa*{g\pa*{x, y}, z}$
		\item $g\pa*{x, f\pa*{y, z}} =_{\qmath} f\pa*{g\pa*{x, y}, g\pa*{x, z}}$
	\end{enumerate}
\end{lemma}

\begin{proof}
	Las demostraciones de los primeros cuatro puntos son anﾃ｡logas a lo que se realiza en las demostraciones del teorema \ref{lemma: prop_suma_prod_z}. La ﾃｺnica necesaria a demostrar es la del ﾃｺltimo punto que es la distributividad. 
	
	Sean $n, p, r\in\zmath$ y $m, q, s\in\zmath[+]$ tales que $x =\mlist{n, m}$, $y =\mlist{p, q}$, $z =\mlist{r, s}$, entonces obtenemos lo siguiente usando el lema \ref{lemma: cancelacion_cociente}:
	\begin{align*}
		g\pa*{x, f\pa*{y, z}} &= g\pa*{x,\mlist{ps + qr, qs}}\\
		&=\mlist{n\pa*{ps + qr}, m\pa*{qs}}\\
		&=\mlist{nps + nqr, mqs}\\
		&=_{\qmath}\mlist{m\pa*{nps + nqr}, m^{2}qs}\\
		&=\mlist{\pa*{np}\pa*{ms} +\pa*{mq}\pa*{nr}, \pa*{mq}\pa*{ms}}\\
		&= f\pa*{\mlist{np, mq},\mlist{nr, ms}}\\
		&= f\pa*{g\pa*{x, y}, g\pa*{x, z}}
	\end{align*}
\end{proof}

\begin{lemma}
	\label{lemma: 0_1_racionales}
	Para cada $n\in\zmath[+]$ se satisface que $\mlist{0, 1} =_{\qmath}\mlist{0, n}$ y $\mlist{1, 1} =_{\qmath}\mlist{n, n}$.
\end{lemma}

\begin{proof}
	Se sigue directamente de las igualdades $0\cdot n = 0$ y $n = n$.
\end{proof}

Podemos entonces ahora definir las operaciones de suma y producto en $\qmath$ de manera coherente.

\begin{definition}[Suma y producto en $\qmath$]
	Se define la \emph{suma} y \emph{producto} en $\qmath$ como las siguientes funciones:
	\[
	\begin{NiceArray}{cc}
		+_{\qmath}:\qmath[2]\to\qmath & \cdot_{\qmath}:\qmath[2]\to\qmath\\
		\pa*{x, y}\mapsto Q\pa*{f\pa*{x, y}} &\pa*{x, y}\mapsto Q\pa*{g\pa*{x, y}}
	\end{NiceArray}
	\]
	donde $f$ y $g$ son las funciones de la definiciﾃｳn \ref{def: suma_producto_q_aux}.
\end{definition}

Se puede entonces probar una propiedad fundamental la cual es que $\qmath$ es un campo.

\begin{theorem}
	El conjunto de nﾃｺmeros racionales $\qmath$ es un campo con las operaciones $+_{\qmath}$ y $\cdot_{\qmath}$.
\end{theorem}

\begin{proof}
	Solo es necesario demostrar la existencia de neutros e inversos (aditivos y multiplicativos) en $\qmath$, pues las otras propiedades de campo son anﾃ｡logas a las de la demostraciﾃｳn del teorema \ref{theo: enteros_anillo}.
	
	\begin{enumerate}
		\item Neutro aditivo
		
		Sea $x =\mlist{p, q}\in\qmath$ para algunos $p\in\zmath$ y $q\in\zmath[+]$. Entonces si definimos $0_{\qmath}\coloneqq Q\pa*{\mlist{0, 1}}$, se obtiene lo siguiente:
		\begin{align*}
			x +_{\qmath} 0_{\qmath} &= Q\pa*{f\pa*{x, 0_{\qmath}}}\\
			&= Q\pa*{f\pa*{\mlist{p, q}, Q\pa*{\mlist{0, 1}}}}\\
			&= Q\pa*{f\pa*{p, q},\mlist{0, 1}}\\
			&= Q\pa*{p\cdot 1 + q\cdot 0, q\cdot 1}\\
			&= Q\pa*{\mlist{p, q}}\\
			&= Q\pa*{x} = x
		\end{align*}
		\item Neutro multiplicativo
		
		Sea $x =\mlist{p, q}\in\qmath$ para algunos $p\in\zmath$ y $q\in\zmath[+]$. Entonces si definimos $1_{\qmath}\coloneqq Q\pa*{\mlist{1, 1}}$, se obtiene lo siguiente:
		\begin{align*}
			x \cdot_{\qmath} 1_{\qmath} &= Q\pa*{g\pa*{x, 1_{\qmath}}}\\
			&= Q\pa*{g\pa*{\mlist{p, q}, Q\pa*{\mlist{1, 1}}}}\\
			&= Q\pa*{g\pa*{p, q},\mlist{1, 1}}\\
			&= Q\pa*{\mlist{p, q}}\\
			&= Q\pa*{x} = x
		\end{align*}
		
		\item Inverso aditivo
		
		Sea $x =\mlist{p, q}\in\qmath$ para algunos $p\in\zmath$ y $q\in\zmath[+]$. Entonces si definimos $-x\coloneqq Q\pa*{\mlist{-p, q}}$, se obtiene lo siguiente:
		\begin{align*}
			x +_{\qmath} \pa*{-x} &= Q\pa*{f\pa*{x, -x}}\\
			&= Q\pa*{f\pa*{\mlist{p, q}, Q\pa*{\mlist{-p, q}}}}\\
			&= Q\pa*{f\pa*{p, q},\mlist{-p, q}}\\
			&= Q\pa*{pq + q\pa*{-p}, q^{2}}\\
			&= Q\pa*{\mlist{0, q^{2}}}\\
			&= Q\pa*{\mlist{0, 1}} = 0_{\qmath}
		\end{align*}
		
		\item Inverso multiplicativo
		
		Sea $x =\mlist{p, q}\in\qmath$ para algunos $p\in\zmath$ y $q\in\zmath[+]$. Si $x\neq 0_{\qmath}$, entonces $p\neq 0$, luego tenemos dos casos: $p > 0$ o $p < 0$.
		
		Si $p > 0$, entonces definimos $x^{-1}\coloneqq Q\pa*{\mlist{q, p}}$. Luego:
		\begin{align*}
			x \cdot_{\qmath} x^{-1} &= Q\pa*{g\pa*{x, x^{-1}}}\\
			&= Q\pa*{g\pa*{\mlist{p, q}, Q\pa*{\mlist{q, p}}}}\\
			&= Q\pa*{g\pa*{p, q},\mlist{q, p}}\\
			&= Q\pa*{\mlist{pq, pq}}\\
			&= Q\pa*{\mlist{1, 1}} = 1_{\qmath}
		\end{align*}
		
		En cambio, si $p < 0$, entonces definimos $x^{-1}\coloneqq Q\pa*{\mlist{-q, -p}}$. Luego:
		\begin{align*}
			x \cdot_{\qmath} x^{-1} &= Q\pa*{g\pa*{x, x^{-1}}}\\
			&= Q\pa*{g\pa*{\mlist{p, q}, Q\pa*{\mlist{-q, -p}}}}\\
			&= Q\pa*{g\pa*{p, q},\mlist{-q, -p}}\\
			&= Q\pa*{\mlist{-pq, -pq}}\\
			&= Q\pa*{\mlist{1, 1}} = 1_{\qmath}
		\end{align*}
		
		En cualquier caso existe $x^{-1}\in\qmath$ tal que $x\cdot_{\qmath}x^{-1} = 1_{\qmath}$
	\end{enumerate}
	Por lo tanto, el conjunto $\qmath$ es un campo con las operaciones $+_{\qmath}$ y $\cdot_{\qmath}$.
\end{proof}

Para terminar con la construcciﾃｳn de los racionales definamos el orden en ellos. Sean $x = n / m$ y $y = p / q$. Puesto que $m$ y $p$ se toman como enteros positivos, entonces se tiene que $x < y$ si y solo si $n / m < p / q$, o equivalentemente $nq < pm$. Entonces de esta manera podemos ahora caracterizar el orden de $\qmath$ en $\rca$.

\begin{definition}[Orden en $\qmath$]
	Se define la \emph{relaciﾃｳn de orden} en $\qmath$ como la relaciﾃｳn binaria $<_{\qmath}\subseteq\qmath[2]$ tal que para cada $x =\mlist{n, m}, y =\mlist{p, q}\in\qmath$ para algunos $n, m, p, q\in\zmath$:
	\[
		x <_{\qmath} y\iff nq < pm
	\]
\end{definition}

\begin{theorem}
	La relaciﾃｳn $<_{\qmath}$ es un orden lineal.
\end{theorem}

\begin{proof}
	Sean $x =\mlist{n, m}, y =\mlist{p, q}, z =\mlist{r, s}\in\qmath$ para algunos $n, p, r\in\zmath$ y $m, q, s\in\zmath[+]$. Mostremos entonces que $<_{\qmath}$ es irreflexiva, transitiva y tricotﾃｳmica.
	
	\begin{enumerate}
		\item Irreflexividad
		
		Puesto que el orden en $\zmath$ es irreflexiva, sabemos entonces que $\neg\pa*{nm < nm}$, es decir, $\neg\pa*{x <_{\qmath} x}$.
		
		\item Transitividad
		
		Supongamos que $x <_{\qmath} y$ y $y <_{\qmath} z$, entonces $nq < pm$ y $ps < rq$. Puesto que $s > 0$, entonces $nqs < pms$. Como $m > 0$, entonces $psm < rqm$. Por transitividad del orden en $\zmath$ tenemos que $nqs < rqm$, y como $q > 0$, entonces $ns < rm$. Por lo tanto, $x <_{\qmath} z$.
		
		\item Tricotomﾃｭa
		
		Por la tricotomﾃｭa en $\zmath$ sabemos que se satisface una y solo una de las siguientes afirmaciones: $nq = pm$, $nq < pm$ o $nq > pm$.
		
		Si $nq = pm$, entonces $x =_{\qmath} y$, y puesto que $x, y\in\qmath$, entonces $x = y$.
		
		Si $nq < pm$, entonces $x <_{\qmath} y$; y si $nq > pm$, entonces $y <_{\qmath} x$.
	\end{enumerate}
	Por lo tanto, $<_{\qmath}$ es un orden lineal.
\end{proof}

Finalmente podemos demostrar igualmente que podemos encajar $\zmath$ en $\qmath$ mediante el encaje $n\mapsto Q\pa*{\mlist{n, 1}}$, pues se interpreta que $n = n / 1$.

\begin{theorem}[Encaje de $\zmath$ en $\qmath$]
	Defﾃｭnase la funciﾃｳn $Z:\zmath\to\qmath$ tal que para cada $n\in\zmath$, $Z\pa*{n}\coloneqq Q\pa*{\mlist{n, 1}}$. Entonces $Z$ es inyectiva y satisface las siguientes propiedades:
	\begin{enumerate}
		\item Para cada $n, m\in\zmath$, $Z\pa*{n + m} = Z\pa*{n} +_{\qmath} Z\pa*{m}$.
		\item Para cada $n, m\in\zmath$, $Z\pa*{nm} = Z\pa*{n} \cdot_{\qmath} Z\pa*{m}$.
		\item Para cada $n, m\in\zmath$, $n < m$ si y solo si $Z\pa*{n} <_{\qmath} Z\pa*{m}$.
	\end{enumerate}
\end{theorem}

\begin{proof}
	Sean $p, q\in\zmath$, entonces:
	\begin{enumerate}
		\item Inyectividad de $Z$
		
		Supongamos que $Z\pa*{p} = Z\pa*{q}$, entonces se tiene que $Q\pa*{\mlist{p, 1}} = Q\pa*{\mlist{q, 1}}$, lo cual significa que $\mlist{p, 1} =_{\qmath}\mlist{q, 1}$, es decir que $p\cdot 1 = q\cdot 1$, o equivalentemente, $p = q$. Por lo tanto, $Z$ es inyectiva.
		
		\item Conservaciﾃｳn de la suma
		
		Se tiene lo siguiente:
		\begin{align*}
			Z\pa*{p + q} &= Q\pa*{\mlist{p + q, 1}}\\
			&= Q\pa*{f\pa*{\mlist{p, 1},\mlist{q, 1}}}\\
			&= Q\pa*{f\pa*{Q\pa*{\mlist{p, 1}}, Q\pa*{\mlist{q, 1}}}}\\
			&= Q\pa*{f\pa*{Z\pa*{n}, Z\pa*{m}}}\\
			&= Z\pa*{n} +_{\qmath} Z\pa*{m}
		\end{align*}
		
		\item Conservaciﾃｳn del producto
		
		Se tiene lo siguiente:
		\begin{align*}
			Z\pa*{pq} &= Q\pa*{\mlist{pq, 1}}\\
			&= Q\pa*{g\pa*{\mlist{p, 1},\mlist{q, 1}}}\\
			&= Q\pa*{g\pa*{Q\pa*{\mlist{p, 1}}, Q\pa*{\mlist{q, 1}}}}\\
			&= Q\pa*{g\pa*{Z\pa*{p}, Z\pa*{q}}}\\
			&= Z\pa*{n} \cdot_{\qmath} Z\pa*{m}
		\end{align*}
		
		\item Conservaciﾃｳn del orden
		
		Sean $Z\pa*{p} =\mlist{n, m}$ y $Z\pa*{q} =\mlist{r, s}$ para algunos $n, r\in\zmath$ y $m, s\in\zmath[+]$, entonces sabemos que $pm = n$ y $qs = r$ por definiciﾃｳn de $Z$, entonces:
		\begin{align*}
			p < q &\iff pm < qm\\
			&\iff n < mq\\
			&\iff ns < mqs\\
			&\iff ns < mr\\
			&\iff \mlist{n, m} <_{\qmath}\mlist{r, s}\\
			&\iff Z\pa*{p} <_{\qmath} Z\pa*{q}
		\end{align*}
	\end{enumerate}
\end{proof}

\subsection{Los nﾃｺmeros reales}

Con los racionales podemos entonces definir lo que significa ser un nﾃｺmero real. Sin embargo, este tipo de nﾃｺmeros tiene una peculiaridad, pues estos no son un conjunto de $\rca$, pues cualquier construcciﾃｳn de los reales implica que estos son incontables, por lo que estos no pueden codificarse como un subconjunto de $\nmath$. Aunque puedan existir modelos incontables de $\rca$, su estructura interna es \emph{contable}, por lo que es imposible codificar a los reales como un conjunto tal cual en $\rca$.
